#include <iostream>
using namespace std;
#define MAX 100

int timNhiPhan(int a[], int l, int r, int x)
{
	if (r>=l) {
		int mid=(l+r)/2;
		if(a[mid]>x) {
			return timNhiPhan(a, l, mid-1, x);
		}
		if(a[mid]<x) {
			return timNhiPhan(a, mid+1, r, x);
		}
		if(a[mid]==x) {
			return mid;
		}
	}
	return -1;
}
	
int main()
{
	int i, x, n;
		cout <<"Nhap vao so phan tu, n=";
		cin >>n;	
	int a[MAX];
	for(i=1; i<n+1; i++) {
		cout <<"Nhap phan tu thu " <<i <<": ";
		cin >>a[i];
	}
		cout <<"\nNhap vao so can tim trong day: x=";
		cin >>x;
	int kq=timNhiPhan(a, 0, n-1, x);
	if (kq==-1)
    	cout <<"So can tim khong xuat hien trong day!";
  	else
    	cout <<"So can tim xuat hien tai chi so " <<kq;
  return 0;
}
